97 // Produce a permitted_operations bitfield with all permission set to true, per the PDF spec
98 constexpr auto make_permitted_operations_all( void )
99 {
100  auto result = ~permitted_operations();
101 
102  // These two bits are required to be off
103  result.set( permitted_operation::unused_0, false );
104  result.set( permitted_operation::unused_1, false );
105  return result;
106 }
