58 template<typename F>
59 void enumerate_structure_leaf_nodes_helper(
60  document& doc,
61  content_range cr,
62  structure_ptr const& str,
63  const structure_node& node,
64  esln_state& state,
65  F&& f )
66 {
67  tetraphilia_assert( cr.m_from.m_pageID == cr.m_to.m_pageID );
68  size_t num_children = node.NumChildren();
69  if( !num_children )
70  {
71  auto this_range = node.GetRange()->m_contentRange;
72  switch( state )
73  {
74  case esln_state::before:
75  // Special case: if the client provided the special "start of page" sentinel content point,
76  // then whatever structure node we hit first is the place to start.
77  if ( cr.m_from == content_point::MakeContentPointAtStartOfPage( cr.m_from.m_pageID ) )
78  cr.m_from = this_range.m_from;
79  if ( this_range.Contains( cr.m_from ) )
80  {
81  state = esln_state::inside;
82  this_range.m_from = cr.m_from;
83  if ( this_range.Contains( cr.m_to ) )
84  this_range.m_to = cr.m_to;
85  f( str, &node, this_range );
86  if ( this_range.Contains( cr.m_to ) )
87  state = esln_state::after;
88  }
89  break;
90  case esln_state::inside:
91  if ( this_range.Contains( cr.m_to ) )
92  {
93  this_range.m_to = cr.m_to;
94  f( str, &node, this_range );
95  state = esln_state::after;
96  }
97  else
98  f( str, &node, this_range );
99  break;
100  default:
101  tetraphilia_assert( state == esln_state::after );
102  return;
103  }
104  }
105  else if ( state != esln_state::after )
106  {
107  for( const auto& kv : node.GetChildList() )
108  enumerate_structure_leaf_nodes_helper( doc, cr, str, *kv.second, state, std::forward<F>( f ) );
109  }
110 }

112 template<typename F>
113 void enumerate_structure_leaf_nodes(
114  document& doc,
115  content_range const& cr,
116  content_order order,
117  F&& f )
118 {
119  auto start_page = std::max<PageIndex>( 0, get_page_num( doc, cr.m_from.m_pageID ) );
120  auto stop_page = std::min<PageIndex>( get_page_num( doc, cr.m_to.m_pageID ), get_page_count( doc ) ) + 1;
121 
122  for ( auto page = start_page; page < stop_page; ++page )
123  {
124  auto pid = get_page_id( doc, page );
125  auto range = content_range::MakeContentRangeForPage( pid );
126 
127  // If we're on the first page in the range, clip to the range's start
128  if ( cr.m_from.m_pageID == pid )
129  range.m_from = cr.m_from;
130 
131  // If we're on the last page in the range, clip to the range's end
132  if ( cr.m_to.m_pageID == pid )
133  range.m_to = cr.m_to;
134 
135  esln_state state = esln_state::before;
136  auto structure = get_structure(doc, pid, order);
137  auto root = structure->GetRoot();
138  if ( root->NumChildren() )
139  {
140  enumerate_structure_leaf_nodes_helper( doc, range, structure, *structure->GetRoot(), state, std::forward<F>( f ) );
141  }
142  else
143  std::forward<F>( f )( {}, nullptr, range );
144  }
145 }

319 template<typename ContentRangeIterator, typename MarkHandler>
320 void roc_text_dl_consumer<ContentRangeIterator, MarkHandler>::check_current_content_range( void )
321 {
322  if ( !m_skip_inclusion_check )
323  {
324  auto cr = std::get<2>( *m_range_iterator );
325  ct::ContentPointLessThan_SamePageDocumentOrder order;
326 
327  auto ccp = this->m_curContentPoint;
328  assert( ccp.m_pageID == cr.m_from.m_pageID && ccp.m_pageID == cr.m_to.m_pageID );
329  tetraphilia_assert( !ccp.m_isAfter );
330  ccp.m_isAfter = true;
331 
332  // If we're past the end of the current content range, advance to the next one
333  if ( t4::ge( ccp, cr.m_to, order ) )
334  {
335  m_found_end_of_range = true;
336  ++m_range_iterator;
337  using mark_handler_default::next_range;
338  next_range( m_mark_handler );
339 
340  // If we've consumed all the ranges, or the next range starts before the current point, we need to restart the enumeration
341  if ( m_range_iterator == m_range_end )
342  this->StopConsumption();
343  else
344  {
345  m_current_range = std::get<2>( *m_range_iterator );
346  if ( m_current_range.m_to.m_pageID != ccp.m_pageID || t4::lt( std::get<2>( *m_range_iterator ).m_from, ccp, order ) )
347  this->StopConsumption();
348  }
349  }
350  }

353 template<typename ContentRangeIterator, typename MarkHandler>
354 bool roc_text_dl_consumer<ContentRangeIterator, MarkHandler>::HandleShow_Begin(
355  bool hasValidUCM,
356  bool isVisible )
357 {
358  tetraphilia_assert( !m_skip_inclusion_check );
359  auto sn = std::get<1>( *m_range_iterator );
360  auto attrib = sn ? sn->GetAttributes() : nullptr;
361 
362  bool goDeeper = false;
363  using mark_handler_default::supports_feature;
364  if ( !attrib || !attrib->IsDropShadow() || supports_feature( m_mark_handler, te::kDropShadows ) )
365  {
366  // Create a content range spanning all the strings in this show and see if it is completely
367  // contained by the current range
368  content_range thisShowRange( this->m_curContentPoint, this->m_curContentPoint );
369  thisShowRange.m_to.m_stringNum = std::numeric_limits<size_t>::max();
370 
371  ct::ContentPointLessThan_DocumentOrder order;
372  if ( t4::gt( thisShowRange.m_to, m_current_range.m_from, order ) )
373  {
374  m_skip_inclusion_check = m_current_range.Contains( thisShowRange );
375  using mark_handler_default::handle_show_begin;
376  goDeeper = handle_show_begin(
377  m_mark_handler,
378  make_enumeration_state(),
379  *this->m_pShowInfo,
380  hasValidUCM,
381  isVisible && ( !attrib || attrib->IsVisible() ) );
382  }
383  }
384 
385  return goDeeper;
386 }

581 /*
582  Given a sequence of content ranges (delimited by start/stop), a document and a content order, create a generator that
583  enumerates all the content in the document that falls "inside" that content range, ordered according to the specified
584  ordering. Each individual reported content range is a z-ordered range.
585 */
586 template<typename Iterator>
587 structure_leaf_node_range make_structure_leaf_node_enumerator(
588  document& doc, content_order order,
589  const Iterator& start, const Iterator& stop )
590 {
591  auto gen = [&doc, order, start, stop]( auto& push )
592  {
593  for ( auto it = start; it != stop; ++it )
594  {
595  enumerate_structure_leaf_nodes( doc, *it, order, [&push]( structure_ptr s, const structure_node* node, const content_range& subrange )
596  {
597  push( std::make_tuple( std::move( s ), node, subrange ) );
598  } );
599  }
600  };
601 
602  #if T4_SUPPORT_COROUTINES
603  return structure_leaf_node_range( std::move( gen ) );
604  #else
605  // If we can't do coroutines, just play the entire sequence into a vector
606  structure_leaf_node_range result;
607  auto push = [&]( structure_leaf_node_element e ){ result.push_back( std::move( e ) ); };
608  gen( push );
609  return result;
610  #endif
611 }

622  document& doc,
623  Iterator& it, Iterator const& stop,
624  MarkHandler&& mark_handler,

626 {
627  while ( it != stop )

631  tetraphilia_assert( pid == cr.m_to.m_pageID );
632

634  namespace tparams = text_extract_params;
