41 /*
42  Enumerated type controlling when a paint request is scheduled.
43 */
44 enum class update_timing
45 {
46  /*
47  Paint requests scheduled for "right now" will be _scheduled_
48  immediately. In practice this means that if the specified page
49  is currently being painted (and that painting does not completely
50  subsume the new request) it will be interrupted.
51  */
52  right_now,
53 
54  /*
55  Paint requests scheduled for "after_next_incremental_update" will
56  be scheduled immediately if no thread is currently working on
57  that same page. If there _is_ a thread painting that page, it
58  will not restart until the next incremental update (which by default
59  happen every 500 msec).
60 
61  If you need to call set_page_visible/set_page_invisible at frame-rate
62  frequencies, please pass this value for the update_timing parameter.
63  */
64  after_next_incremental_update
65 };

67 struct paint_job_params
68 {
69  page_num m_page;
70  draw_params m_draw_params;
71  int_rect m_clip_rect;
72  std::chrono::steady_clock::duration m_update_interval;
73 
74  bool subsumes( paint_job_params const& other )
75  {
76  return m_page == other.m_page
77  && m_draw_params == other.m_draw_params
78  && contains( m_clip_rect, other.m_clip_rect );
79  }
80 };

82 /*
83  shared_ptrs to paint_requests are the fundamental queued entity in the
84  pdf_tile_cache. pdf_tile_cache_requesters create them and return them
85  request_impl(). They can be updated, and the update logic is smart about
86  not unnecessarily restarting renderings.
87 
88  paint_request is internally synchronized. The expectation is that implementers
89  of pdf_tile_cache_requester will create and modify paint_requests. The
90  pdf_tile_cache will only query it and respond to changes to its state.
91 */
92 template<typename PDFTCClient>
93 class paint_request : boost::noncopyable
94 {
95  using paint_request_priority = paint_request_priority_t<PDFTCClient>;
96  using tc_client = pdf_detail::tc_client<PDFTCClient>;
97  using tile_grid = tile_cache::tile_grid<tc_client>;
98 
99  // This mutex guards all of paint_request's state
100  mutex mutable m_mutex;
101  paint_job_params m_paint_job_params;
102  boost::optional<boost::variant2::variant<paint_job_params, pdf_detail::cancel_t>> m_deferred_paint_job_params;
103  paint_request_priority m_priority;
104  thread* m_thread;
105  boost::optional<tile_grid> m_completed_grid;
106 
107 public:
108  paint_request( paint_job_params const& pjp, paint_request_priority priority )
109  : m_paint_job_params( std::move( pjp ) ),
110  m_priority( priority ),
111  m_thread( nullptr )
112  {}
113 
114  ~paint_request( void ) {}
115 
116  paint_job_params get_paint_job_params( void ) const
117  {
118  return boost::with_lock_guard( m_mutex, [this]() { return m_paint_job_params; } );
119  }
120 
121  paint_request_priority get_priority( void ) const { return m_priority; }
122 
123  bool available( void ) const
124  {
125  return boost::with_lock_guard( m_mutex, [this]( void )
126  {
127  return m_thread == nullptr && !m_completed_grid;
128  } );
129  }
130 
131  void update( paint_job_params const& pjp, update_timing timing )
132  {
133  boost::with_lock_guard( m_mutex, [this, &pjp, timing]( void )
134  {
135  if ( !m_paint_job_params.subsumes( pjp ) )
136  {
137  if ( m_thread || timing == update_timing::right_now )
138  {
139  m_completed_grid.reset();
140  m_paint_job_params = pjp;
141  m_deferred_paint_job_params.reset();
142  }
143 
144  if ( m_thread )
145  {
146  switch( timing )
147  {
148  case update_timing::right_now:
149  if ( m_thread->get_id() == t4::this_thread::get_id() )
150  {
151  m_thread = nullptr;
152  throw t4::thread_interrupted();
153  }
154  else
155  {
156  m_thread->interrupt();
157  m_thread = nullptr;
158  }
159  break;
160  case update_timing::after_next_incremental_update:
161  m_deferred_paint_job_params = pjp;
162  break;
163  }
164  }
165  }
166 
167  } );
168  }
169 
170  void update_eventually( paint_job_params const& pjp )
171  {
172  boost::with_lock_guard( m_mutex, [this, &pjp]( void )
173  {
174  if ( !m_paint_job_params.subsumes( pjp ) )
175  {
176  if ( m_thread )
177  m_deferred_paint_job_params = pjp;
178  else
179  m_paint_job_params = pjp;
180  }
181  } );
182  }
183 
184  void cancel_eventually( void )
185  {
186  boost::with_lock_guard( m_mutex, [this]( void )
187  {
188  if ( m_thread )
189  m_deferred_paint_job_params = pdf_detail::cancel;
190  } );
191  }
192 
193  bool claim( thread* th )
194  {
195  return boost::with_lock_guard( m_mutex, [this, th]( void )
196  {
197  if ( m_thread == nullptr )
198  {
199  m_thread = th;
200  return true;
201  }
202  return false;
203  } );
204  }
205 
206  void interrupt( void )
207  {
208  boost::with_lock_guard( m_mutex, [this]( void )
209  {
210  if (m_thread)
211  {
212  m_thread->interrupt();
213  m_thread = nullptr;
214  }
215  } );
216  }

219  {
220  typedef void result_type;

239  void consider_restart( t4::thread::id tid )
240  {
241  boost::with_lock_guard( m_mutex, [this, tid]( void )
242  {
243  if ( m_thread && m_thread->get_id() == tid && m_deferred_paint_job_params )
244  visit( deferred_paint_visitor( this ), *m_deferred_paint_job_params );
245  } );
246  }
247 
248  void complete( thread* th, tile_grid tg )
249  {
250  boost::with_lock_guard( m_mutex, [this, th, &tg]( void )
251  {
252  // First make sure we're talking to the thread who is supposed to be working on this.
253  // There could be a valid mismatch under the following conditions:
254  //
255  // Thread A starts to paint page x with params pA
256  // A.interrupt() is called
257  // Thread B starts to paint page x with params pB
258  // Thread B finishes
259  // Thread B calls complete() with finished product at desired params
260  // A completes without checking for interruption, calls complete with finished product at incorrect (stale) params pA
261  if ( th == m_thread )
262  {
263  assert( tg.get_scale() == m_paint_job_params.m_draw_params.m_scale );
264  assert( !m_completed_grid );
265  m_completed_grid = std::move( tg );
266  m_thread = nullptr;
267  }
268  } );
269  }
270 };
