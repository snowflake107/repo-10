40 /*
41  A draw_client is used to handle progress/completion notification, and override annotation
42  rendering if desired.
43 
44  To make your own draw_client, write a class with any of the three callbacks below implemented as
45  free functions. For example:
46 
47  class my_draw_client
48  {
49  std::ostream& m_stm;
50 
51  public:
52  my_draw_client( std::ostream& stm ) : m_stm( stm ) {}
53 
54  friend void mark( my_draw_client const& ths, size_t obj_count, int_rect const&, byte_transparency_group& )
55  {
56  ths.m_stm << obj_count << " objects drawn so far.\n";
57  }
58 
59  friend void drawing_finished( my_draw_client const& ths, size_t obj_count, byte_transparency_group& group, t4::exception_ptr const& e )
60  {
61  ths.m_stm << "finished.\n";
62  }
63 
64  // You don't have to provide implementations of every callback. For example, we're leaving override_annotation_appearance
65  // out here, so a default implementation will be selected. But if you wanted to provide an implementation of that
66  // callback, you could uncomment this function.
67  #if 0
68  friend boost::optional<dictionary> override_annotation_appearance( my_draw_client const& ths, dictionary const& appearance_dict )
69  {
70  // put implementation here
71  }
72  #endif
73  };
74 
75  You can create a draw_client from an instance of my_draw_client thus:
76 
77  draw_client dc = make_draw_client( my_draw_client( std::cout ) );
78 
79  draw_client is a regular type. You can pass it to draw_page using the dparams::draw_client parameter tag. Its callbacks will
80  not be called outside the lifetime of an invocation of draw_page to which it is passed.
81 */
82 class draw_client
83 {
84  std::shared_ptr<draw_detail::draw_client_impl_base> m_impl;
85 
86 public:
87  draw_client( void );
88 
89  void mark( size_t obj, int_rect const& bounds, byte_transparency_group& group ) const;
90 
91  void drawing_finished( size_t obj_count, byte_transparency_group& group, t4::exception_ptr const& e ) const;
92 
93  boost::optional<dictionary> override_annotation_appearance( dictionary const& appearance_dict ) const;
94 
95 private:
96  template<typename Impl>
97  friend draw_client make_draw_client( Impl impl );
98 
99  draw_client( std::shared_ptr<draw_detail::draw_client_impl_base> impl );
100 };

180 /*
181  This data structure describes all the special, non-required parameters to a draw
182  (or required, but with a default value).
183 
184  For example, to create a draw_parameters object that specifies a greek_threshold of 3:
185 
186  namespace dparams = t4::pdf::dparams;
187  using t4::pdf::draw_parameters_desc;
188  using t4::parameter;
189 
190  auto params = draw_parameters_desc( parameter<dparams::greek_threshold>( 3 ) );
191  draw_page( pixels, scan_line_stride, doc, pg, base_page_xform, page_to_device_xform, bounds, params );
192 
193  You can modify an existing draw_parameters object using the += operator:
194 
195  params += parameter<dparams::optional_border_color>( 0xC0000000 );
196 */
197 constexpr auto draw_parameters_desc = parameter_description(
198  optional_parameter<dparams::reflow_params>,
199  optional_parameter<dparams::deprecated::reflow_layout>,
200  optional_parameter<dparams::mcids>,
201  required_parameter<dparams::greek_threshold>( 2 ),
202  required_parameter<dparams::draw_client>( lazy_default( get_null_draw_client ) ),
203  required_parameter<dparams::required_border_color>( 0x80000000 ),
204  required_parameter<dparams::optional_border_color>( 0x80000000 ),
205  required_parameter<dparams::display_event_stream>( lazy_default( get_event_ignorer ) ),
206  required_parameter<dparams::draw_annotations>( yes<draw_annotations_flag> ),
207  optional_parameter<dparams::optional_content>,
208  required_parameter<dparams::result_user_space_bbox>( nullptr_to<real_rect> ) );
