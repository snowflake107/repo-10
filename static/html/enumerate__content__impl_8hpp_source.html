87 template<typename F>
88 void enumerate_paragraphs( t4::pdf::document& doc, t4::pdf::page_num page, F&& f, content_order order )
89 {
90  using namespace t4::literals;
91 
92  /*
93  For given page, we intend to receive callbacks for each character in structure
94  order. This will help in accumulating "current" line and paragraph state which
95  will be updated and delivered to client when a paragraph end is detected. To
96  detect paragraph end, we maintain a list of paragraph content ranges and a
97  sliding iterator to the current paragraph within this list.
98  */
99  auto pid = t4::pdf::get_page_id( doc, page );
100  content_range page_range = { content_point::MakeContentPointAtStartOfPage( pid ), content_point::MakeContentPointAtEndOfPage( pid ) };
101 
102  // Local state to build up via character-level callbacks
103  unicode_string str;
104  paragraph para;
105 
106  auto structure = get_structure( doc, pid, order );
107 
108  // Maintain iterator to current paragraph
109  auto para_list = structure->GetPageParagraphList();
110  auto current_para_iter = para_list.begin();
111  auto get_next_para = [&]() -> boost::optional<content_range>
112  {
113  if ( current_para_iter != para_list.end() )
114  {
115  return *current_para_iter++;
116  }
117  return boost::none;
118  };
119 
120  boost::optional<content_range> current_para = get_next_para();
121  if (!current_para)
122  current_para = page_range;
123 
124  // Receive character geometry
125  boost::optional<std::pair<t4::real_point, t4::real_point>> leading, trailing;
126 
127  auto close_line = [&]( void )
128  {
129  // Found a line
130  if ( !str.empty() )
131  {
132  auto line = std::make_pair( std::move(str), make_pair( *leading, *trailing ) );
133  str.clear();
134  para.push_back( line );
135  }
136  };
137 
138  auto close_paragraph = [&]( void )
139  {
140  // Found a paragraph
141  if ( !para.empty() )
142  {
143  // Report to client
144  f( para );
145 
146  // Update state for next para
147  para.clear();
148  current_para = get_next_para();
149  }
150  };
151 
152  auto accumulate_line = [&]( std::uint32_t u, extracted_glyph g )
153  {
154  // Update current line
155  if ( leading )
156  {
157  if ( g.StartsNewLine() )
158  {
159  assert( trailing );
160  close_line();
161  leading = {};
162 
163  if ( current_para && !current_para->Contains( g.ContentPoint() ) )
164  {
165  close_paragraph();
166  }
167  }
168  }
169 
170  if ( !leading )
171  leading = g.GetLeadingEdge();
172 
173  trailing = g.GetTrailingEdge();
174 
175  // Push character into current line
176  str.push_back( u );
177  };
178 
179  // Iterate through each character (in structure order) and deduce line/paragraph
180  // info via received callbacks
181  t4::pdf::enumerate_content( doc, page_range, std::move(accumulate_line), order );
182 
183  // Close pending state
184  close_line();
185  close_paragraph();
186 }
187
